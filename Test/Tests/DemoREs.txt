[re_IntegerNumber]

=^(?# BOL marker)[+\-]?\d+$(?# EOL marker)

ModifierStr=x
SubstitutionTemplate=
-The r.e. above expects only digits in the string,
= because simbols of start and end of string
= ('^' and '$').
=So, you must remove all non digits symbols in
=input string or symbols '^' and '$' in r.e.
String4Replace=<here was digit>
